\section{Perfis de Usuário}

\subsection{Administrador (Admin)}
\begin{itemize}
    \item Acesso total ao sistema
    \item Gerencia usuários e permissões
    \item Visualiza todos os setores
\end{itemize}

\subsection{Gestor (Manager)}
\begin{itemize}
    \item Gerencia documentos do seu setor
    \item Aprova/rejeita processos
    \item Visualiza relatórios
\end{itemize}

\subsection{Operador (Operator)}
\begin{itemize}
    \item Cria e edita documentos próprios
    \item Cria processos
    \item Não pode aprovar ou excluir
\end{itemize}

\subsection{Visitante (Viewer)}
\begin{itemize}
    \item Apenas visualização
    \item Pode usar o chat de IA
    \item Sem permissão de edição
\end{itemize}

\section{Regras de Negócio}

\begin{enumerate}
    \item RN01: Documentos só podem ser publicados por Admin ou Gestor
    \item RN02: Processos rejeitados podem ser reabertos pelo solicitante
    \item RN03: Arquivos devem ter no máximo 50MB
    \item RN04: Sessão expira após 30 minutos de inatividade
    \item RN05: Histórico de alterações deve ser mantido por 2 anos
\end{enumerate}
