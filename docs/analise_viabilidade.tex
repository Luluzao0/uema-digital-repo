\section{Viabilidade Técnica}

\subsection{Tecnologias Utilizadas}
\begin{itemize}
    \item \textbf{Frontend:} React 18, TypeScript, Vite, Tailwind CSS
    \item \textbf{Backend:} Supabase (PostgreSQL + Auth + Storage)
    \item \textbf{IA:} Cohere API (assistente e RAG)
    \item \textbf{Hospedagem:} Vercel (frontend), Supabase Cloud (backend)
\end{itemize}

\subsection{Avaliação}
\begin{itemize}
    \item Todas as tecnologias são gratuitas ou open source
    \item Equipe possui conhecimento em React e TypeScript
    \item Infraestrutura em nuvem não requer manutenção de servidores
\end{itemize}

\textbf{Resultado:} \textit{Viável}
